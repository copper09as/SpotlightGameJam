local GC = CS.GameController

function CharacterJump(character)
    if character.rb == nil then return end
    -- 判断是否在地面
    if character:GroundCheck() then
        -- 设置垂直速度实现跳跃
        local velocity = character.rb.velocity
        velocity.y = character.CharacterData.JumpHeight
        character.rb.velocity = velocity
    end
end
