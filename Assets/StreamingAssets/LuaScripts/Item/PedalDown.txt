function PedalDown(entity)--踏板被踩下
    local data = entity.CommonEntityData --实体通用数据
    local a = 1 --体型判定值
    local key = "OpenTheDoor"

    print(entity.entityPairs.Count)
    if a<=  data.ScaleX*data.ScaleY then
        --对此列表的所有对象进行对应效果触发
        for i, entityPair in ipairs(entity.entityPairs) do
            if entityPair.key == key then
                for j, otherEntity in ipairs(entityPair.entities) do
                    OpenTheDoor(otherEntity)
                end
            end
        end

    end
    --写怎么样触发开门

end
