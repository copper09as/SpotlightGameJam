function PedalDown(entity)--踏板被踩下
    local data = entity.CommonEntityData --实体通用数据
    local a = 1 --体型判定值
    local key = "OpenTheDoor"

    print(entity.entityPairs.Count)
    if a<=  data.ScaleX*data.ScaleY then
		local entities = entity:GetEntities(key)
		local entitiesCount = entities.Count
		for j = 0, entitiesCount - 1 do
			local otherEntity = entities[j]
				
			OpenTheDoor(otherEntity)
		end
    end
    --写怎么样触发开门

end

function OpenTheDoor(entity)
    entity.animator:SetBool("IsOpen",true)
end
