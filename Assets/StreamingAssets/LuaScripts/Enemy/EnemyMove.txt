local DM = CS.DetectionManager--检测管理器

function EnemyMove(entity)

    if entity.rb == nil then
        return
    end

    local rb = entity.rb
    local EnemySpeed = 10

    local velocity = rb.velocity

    local x = DM.MultiRayGroundCheck(--检测左边缘
            entity.col ,
            3,1,"Ground"
    )

    local a = x.isLeftEdge
    local b = x.isRightEdge


    print(velocity)
    if a or b then
    velocity.x = 0
    end

    if a then
        velocity.x = velocity.x + EnemySpeed;
    end

    if b then
        velocity.x = velocity.x - EnemySpeed;
    end

    rb.velocity = velocity


end
